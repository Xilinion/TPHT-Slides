  }

\begin{frame}{TPHT in One Slide}
  \begin{columns}[T,onlytextwidth]
    \column{0.33\textwidth}
      \begin{block}{1. Tiny pointers}
        Replace 64-bit pointers with 8-bit references through a specialized dereference table.
      \end{block}
    \column{0.33\textwidth}
      \begin{block}{2. Key quotienting}
        Store only the key bits that are not already implied by the item's location.
      \end{block}
    \column{0.33\textwidth}
      \begin{block}{3. Two layouts}
        Use the same primitives to make either the smallest table or the fastest compact table.
      \end{block}
  \end{columns}
  \vspace{0.35cm}
  \begin{center}

\begin{frame}{Main Technique 1: Tiny Pointers}
  \begin{columns}[T,onlytextwidth]
    \column{0.47\textwidth}
      \includegraphics[width=\linewidth,height=0.72\textheight,keepaspectratio]{assets/figures/tpimpl_v2.pdf}
    \column{0.49\textwidth}
      \begin{block}{A pointer becomes one byte}
        \begin{itemize}
          \item store objects in a fixed-size \keyterm{dereference table}
          \item allocate using the \keyterm{power of two choices}
          \item encode choice bit + in-bin offset as an 8-bit pointer
          \item dereference is arithmetic once the owner ID is known
        \end{itemize}
      \end{block}
      \vspace{0.25cm}
      \metric{98.2\%}{supported load factor for insertion-only workloads}\par
      \metric{95\%}{supported load factor with alternating inserts/deletes}
  \end{columns}
\end{frame}

\begin{frame}{Main Technique 2: Quotient the Key}
  \begin{columns}[T,onlytextwidth]
    \column{0.42\textwidth}
      \includegraphics[width=\linewidth,height=0.67\textheight,keepaspectratio]{assets/figures/quotient_hash_v3.pdf}
    \column{0.54\textwidth}
      \begin{block}{The location already stores information}
        If a key belongs to quotient group \(q\), the high-order information represented by \(q\) does not need to be stored again.
      \end{block}
      \begin{block}{The systems problem}
        Filters can quotient fingerprints. Hash tables must recover the real key.
      \end{block}
      \begin{block}{TPHT's move}
        Use a one-round Feistel permutation so the key can be quotiented and still decoded.
      \end{block}
      \centering
      \metric{\(\log n\)}{bits saved per key, in the useful regime}
  \end{columns}
\end{frame}

\begin{frame}{Design 1: Chained-TPHT}
  \begin{columns}[T,onlytextwidth]
    \column{0.55\textwidth}
      \includegraphics[width=\linewidth,height=0.67\textheight,keepaspectratio]{assets/figures/tpht_chained_v2.pdf}
    \column{0.41\textwidth}
      \begin{block}{Make chaining small enough to be modern}
        \begin{itemize}
          \item head array stores 1-byte tiny pointers
          \item nodes store next pointer, quotiented key remainder, value
          \item quotient groups are exactly chains
          \item quotienting pays back the pointer overhead
        \end{itemize}
      \end{block}
      \vspace{0.25cm}
      \metric{105.4\%}{space efficiency}\par
      \metric{38.6\%}{mean memory reduction vs. strong baselines}
  \end{columns}
\end{frame}

\begin{frame}{Why Chaining Becomes Good Again}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
      \begin{block}{Conventional chaining}
        \begin{itemize}
          \item 8-byte pointers dominate metadata
          \item head arrays become expensive
          \item pointer chasing hurts locality
        \end{itemize}
      \end{block}
    \column{0.48\textwidth}
      \begin{block}{Chained-TPHT}
        \begin{itemize}
          \item 1-byte head pointers make many buckets cheap
          \item compact head array is more cache-resident
          \item quotienting removes enough key bits to offset links
        \end{itemize}
      \end{block}
  \end{columns}
  \vspace{0.25cm}
  \begin{center}
    \begin{tikzpicture}
      \node[draw=CornellRed,line width=1pt,rounded corners=2pt,inner sep=8pt,text width=0.82\textwidth,align=center] {
        Chained-TPHT turns the oldest collision-resolution idea into a practical succinct hash table with expected \(O(1)\) operations.
      };
    \end{tikzpicture}
  \end{center}
\end{frame}

\begin{frame}{Design 2: Flattened-TPHT}
  \begin{columns}[T,onlytextwidth]
    \column{0.55\textwidth}
      \begin{tikzpicture}[font=\small]

        \node[anchor=west] at (0,4.05) {\textbf{64-byte home block}};
        \node[anchor=west,align=left] at (0,2.25) {up to four home tuples\\plus fingerprints and tiny pointers};
        \node[draw=BlueGray,rounded corners=2pt,minimum width=2.15cm,minimum height=0.85cm] (overflow) at (5.2,1.35) {overflow table};
        \draw[-Latex,thick,BlueGray] ($(block.west)+(5.25,-0.35)$) to[out=-80,in=130] (overflow.north);
        \draw[-Latex,thick,BlueGray] ($(block.west)+(6.1,-0.35)$) to[out=-80,in=60] (overflow.north);
        \node[anchor=west,align=left,text width=6.8cm] at (0,0.35) {
          Most operations inspect one cache line; overflow tuples cost one extra miss rather than a chain walk.
        };
      \end{tikzpicture}
    \column{0.41\textwidth}
      \begin{block}{Spend a little space where latency sees it}
        \begin{itemize}
          \item home blocks fit in one cache line
          \item overflow references are tiny pointers
          \item SIMD fingerprints prune comparisons
          \item queries average about 1.25 cache misses
        \end{itemize}
      \end{block}
      \vspace{0.2cm}
      \metric{83.4\%}{space efficiency}\par
      \metric{310.4M}{average run-phase ops/s}
  \end{columns}
\end{frame}
